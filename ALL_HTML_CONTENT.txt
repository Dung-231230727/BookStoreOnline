<script>
    $(document).ready(function() {
        // Hide header/footer for main content focus
        // Uncomment if needed: $("#header-area, #footer-area").hide();

        // Form submission
        $("#changePasswordForm").on("submit", function(e) {
            e.preventDefault();

            const currentPassword = $("#currentPassword").val();
            const newPassword = $("#newPassword").val();
            const confirmPassword = $("#confirmPassword").val();

            // Validation
            if (!validateForm(currentPassword, newPassword, confirmPassword)) {
                return;
            }

            const submitBtn = $("#submitBtn");
            submitBtn.prop("disabled", true);
            submitBtn.html('<i class="fas fa-spinner fa-spin me-2"></i>Đang cập nhật...');

            // API call
            api.put("/auth/change-password", {
                oldPassword:     currentPassword,
                newPassword:     newPassword,
                confirmPassword: confirmPassword
            })
                .done(function(response) {
                    showAlert("success", "✓ Mật khẩu của bạn đã được cập nhật thành công!", "alertContainer");
                    
                    // Clear form
                    $("#changePasswordForm")[0].reset();
                    $("#passwordStrengthBar").css("width", "0%");

                    // Redirect to profile after 2 seconds
                    setTimeout(function() {
                        layout.render("Users", "Profile");
                    }, 2000);
                })
                .fail(function(jqXHR) {
                    const errorMsg = jqXHR.responseJSON?.message || "Có lỗi xảy ra. Vui lòng thử lại.";
                    showAlert("danger", errorMsg, "alertContainer");
                })
                .always(function() {
                    submitBtn.prop("disabled", false);
                    submitBtn.html('<i class="fas fa-save me-2"></i>Lưu Mật Khẩu Mới');
                });
        });

        // Helper functions
        function validateForm(currentPwd, newPwd, confirmPwd) {
            const alertContainer = $("#alertContainer");

            if (!currentPwd) {
                showAlert("danger", "Vui lòng nhập mật khẩu hiện tại", "alertContainer");
                return false;
            }

            if (!newPwd) {
                showAlert("danger", "Vui lòng nhập mật khẩu mới", "alertContainer");
                return false;
            }

            if (!confirmPwd) {
                showAlert("danger", "Vui lòng xác nhận mật khẩu mới", "alertContainer");
                return false;
            }

            if (newPwd !== confirmPwd) {
                showAlert("danger", "Mật khẩu không khớp", "alertContainer");
                return false;
            }

            if (currentPwd === newPwd) {
                showAlert("danger", "Mật khẩu mới phải khác với mật khẩu hiện tại", "alertContainer");
                return false;
            }

            if (!isStrongPassword(newPwd)) {
                showAlert("danger", "Mật khẩu không đủ mạnh. Vui lòng kiểm tra yêu cầu ở trên", "alertContainer");
                return false;
            }

            return true;
        }

        function isStrongPassword(pwd) {
            return pwd.length >= 8 &&
                /[A-Z]/.test(pwd) &&
                /[a-z]/.test(pwd) &&
                /[0-9]/.test(pwd) &&
                /[!@#$%^&*]/.test(pwd);
        }

        function showAlert(type, message, containerId) {
            const alertHTML = `
                <div class="alert alert-${type === 'danger' ? 'danger' : type} alert-dismissible fade show" role="alert" style="border-radius: 8px;">
                    <i class="fas fa-${type === 'danger' ? 'exclamation-circle' : 'check-circle'} me-2"></i>${message}
                    <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
                </div>
            `;
            $("#" + containerId).html(alertHTML);
            window.scrollTo(0, 0);
        }
    });

    // Toggle password visibility
    function togglePasswordField(fieldId) {
        const field = $("#" + fieldId);
        const type = field.attr("type") === "password" ? "text" : "password";
        field.attr("type", type);
        $(`button[onclick*="${fieldId}"] i`).toggleClass("fa-eye fa-eye-slash");
    }

    // Check password strength
    function checkPasswordStrength() {
        const pwd = $("#newPassword").val();
        let strength = 0;
        let requirements = {
            length: pwd.length >= 8,
            upper: /[A-Z]/.test(pwd),
            lower: /[a-z]/.test(pwd),
            number: /[0-9]/.test(pwd),
            special: /[!@#$%^&*]/.test(pwd)
        };

        // Update requirement badges
        updateRequirement("req-length", requirements.length);
        updateRequirement("req-upper", requirements.upper);
        updateRequirement("req-lower", requirements.lower);
        updateRequirement("req-number", requirements.number);
        updateRequirement("req-special", requirements.special);

        // Calculate strength
        strength = Object.values(requirements).filter(Boolean).length;

        // Update strength bar
        const strengthBar = $("#passwordStrengthBar");
        const strengthText = $("#passwordStrengthText");
        const percentage = (strength / 5) * 100;

        strengthBar.css("width", percentage + "%");

        if (strength <= 2) {
            strengthBar.removeClass("bg-warning bg-success").addClass("bg-danger");
            strengthText.html('<i class="fas fa-times-circle text-danger"></i> Mật khẩu yếu');
        } else if (strength <= 4) {
            strengthBar.removeClass("bg-danger bg-success").addClass("bg-warning");
            strengthText.html('<i class="fas fa-exclamation-circle text-warning"></i> Mật khẩu trung bình');
        } else {
            strengthBar.removeClass("bg-danger bg-warning").addClass("bg-success");
            strengthText.html('<i class="fas fa-check-circle text-success"></i> Mật khẩu mạnh');
        }
    }

    function updateRequirement(elementId, met) {
        const badge = $("#" + elementId);
        if (met) {
            badge.html(badge.html().replace("fa-x", "fa-check").replace("text-danger", "text-success"));
        } else {
            badge.html(badge.html().replace("fa-check", "fa-x").replace("text-success", "text-danger"));
        }
    }

    // Check password match
    function checkPasswordMatch() {
        const newPwd = $("#newPassword").val();
        const confirmPwd = $("#confirmPassword").val();
        const matchMsg = $("#passwordMatchMessage");

        if (!confirmPwd) {
            matchMsg.html("");
            return;
        }

        if (newPwd === confirmPwd) {
            matchMsg.html('<small class="text-success mt-2 d-block"><i class="fas fa-check-circle"></i> Mật khẩu khớp</small>');
        } else {
            matchMsg.html('<small class="text-danger mt-2 d-block"><i class="fas fa-times-circle"></i> Mật khẩu không khớp</small>');
        }
    }
</script>

<script>
    $(document).ready(function() {
        $("#header-area, #footer-area").hide();

        // Form submission
        $("#forgotPasswordForm").on("submit", function(e) {
            e.preventDefault();

            const email = $("#email").val().trim();
            const submitBtn = $("#submitBtn");
            const alertContainer = $("#alertContainer");

            // Validation
            if (!email) {
                showAlert("danger", "Vui lòng nhập email của bạn", alertContainer);
                return;
            }

            // Show loading
            submitBtn.prop("disabled", true);
            submitBtn.html('<i class="fas fa-spinner fa-spin me-2"></i>Đang gửi...');

            // API call
            api.post("/auth/forgot-password", { email: email })
                .done(function(response) {
                    if (response.status === 200) {
                        showAlert("success", "✓ " + response.message, alertContainer);
                        
                        // Clear form
                        $("#forgotPasswordForm")[0].reset();

                        // Redirect to login after 3 seconds
                        setTimeout(function() {
                            layout.render("Auth", "Login");
                        }, 3000);
                    } else {
                        showAlert("danger", response.message || "Có lỗi xảy ra. Vui lòng thử lại.", alertContainer);
                    }
                })
                .fail(function(jqXHR) {
                    const errorMsg = jqXHR.responseJSON?.message || "Email không tồn tại hoặc không hợp lệ";
                    showAlert("danger", errorMsg, alertContainer);
                })
                .always(function() {
                    submitBtn.prop("disabled", false);
                    submitBtn.html('<i class="fas fa-paper-plane me-2"></i>Gửi Hướng Dẫn Đặt Lại');
                });
        });

        // Alert helper
        function showAlert(type, message, container) {
            const alertHTML = `
                <div class="alert alert-${type === 'danger' ? 'danger' : type} alert-dismissible fade show" role="alert">
                    <i class="fas fa-${type === 'danger' ? 'exclamation-circle' : 'check-circle'} me-2"></i>${message}
                    <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
                </div>
            `;
            container.html(alertHTML);
            container.scrollTop();
        }
    });
</script>

(function() {
    // Self-contained edit loader — does NOT override books.loadDetail
    async function loadBookForEdit(isbn) {
        try {
            const res = await api.get(`/sach/${isbn}`);
            const b = res.data || res;

            // Store ISBN on button
            document.getElementById('btn-update-book').dataset.isbn = b.isbn;
            document.getElementById('edit-isbn-display').value = b.isbn;
            document.getElementById('edit-tenSach').value = b.tenSach || '';
            document.getElementById('edit-moTa').value = b.moTaNguNghia || b.moTa || '';
            // soTrang removed
            document.getElementById('edit-trongLuong').value = b.trongLuong || '';
            document.getElementById('edit-namXuatBan').value = b.namXuatBan || '';
            document.getElementById('edit-giaNiemYet').value = b.giaNiemYet || '';
            document.getElementById('edit-price-display').innerText = api.formatCurrency(b.giaNiemYet || 0);
            document.getElementById('edit-anhBia').value = b.anhBia || '';
            if (b.anhBia) document.getElementById('edit-book-preview').src = 'assets/images/' + b.anhBia;
            if (b.ngonNgu) document.getElementById('edit-ngonNgu').value = b.ngonNgu;

            // After dropdowns are populated, select correct values
            setTimeout(() => {
                if (b.danhMuc) document.getElementById('maDanhMuc').value = b.danhMuc.maDanhMuc;
                if (b.nxb)     document.getElementById('maNxb').value = b.nxb.maNxb;
            }, 700);
        } catch (e) {
            api.showToast('Không thể tải dữ liệu sách: ' + e.message, 'error');
        }
    }

    // Read ISBN from layout.current.id
    const isbn = (window.layout && window.layout.current) ? window.layout.current.id : null;
    if (isbn) {
        loadBookForEdit(isbn);
    }
})();

<script>
    let selectedBookId = null;
    let selectedRating = 0;

    $(document).ready(function() {
        // Get book ID from URL params if available
        const urlParams = new URLSearchParams(window.location.search);
        const bookIdParam = urlParams.get('bookId');

        if (bookIdParam) {
            selectedBookId = bookIdParam;
            loadBookInfo();
            $("#bookSelectionGroup").hide();
            $("#selectedBookInfo").show();
            $("#changeBookBtn").show();
        } else {
            loadAvailableBooks();
            $("#bookSelectionGroup").show();
            $("#selectedBookInfo").hide();
        }

        // Star rating interaction
        $("#starRating .star").on("click", function() {
            selectedRating = $(this).data("rating");
            $("#rating").val(selectedRating);
            updateStarDisplay(selectedRating);
            updateRatingText();
        });

        $("#starRating .star").on("mouseover", function() {
            const hoverRating = $(this).data("rating");
            updateStarDisplay(hoverRating);
        });

        $("#starRating").on("mouseout", function() {
            updateStarDisplay(selectedRating);
        });

        // Character counters
        $("#title").on("keyup", function() {
            $("#titleCount").text($(this).val().length + "/100");
        });

        $("#content").on("keyup", function() {
            $("#contentCount").text($(this).val().length + "/1000");
        });

        // Form submission
        $("#reviewForm").on("submit", function(e) {
            e.preventDefault();

            if (!validateForm()) return;

            const highlights = [];
            $(".highlight-checkbox:checked").each(function() {
                highlights.push($(this).val());
            });

            const reviewData = {
                bookId: selectedBookId,
                rating: selectedRating,
                title: $("#title").val(),
                content: $("#content").val(),
                highlights: highlights
            };

            const submitBtn = $("#submitBtn");
            submitBtn.prop("disabled", true);
            submitBtn.html('<i class="fas fa-spinner fa-spin me-2"></i>Đang gửi...');

            // API call
            api.post("/reviews/create", reviewData)
                .done(function(response) {
                    showAlert("success", "✓ Cảm ơn! Đánh giá của bạn đã được gửi thành công. Sẽ được xuất bản sau khi kiểm duyệt.", "alertContainer");
                    
                    setTimeout(function() {
                        layout.render("Books", "Details", selectedBookId);
                    }, 2000);
                })
                .fail(function(jqXHR) {
                    const errorMsg = jqXHR.responseJSON?.message || "Có lỗi xảy ra. Vui lòng thử lại.";
                    showAlert("danger", errorMsg, "alertContainer");
                })
                .always(function() {
                    submitBtn.prop("disabled", false);
                    submitBtn.html('<i class="fas fa-paper-plane me-2"></i>Gửi Đánh Giá');
                });
        });
    });

    function loadAvailableBooks() {
        api.get("/books/list?limit=100")
            .done(function(response) {
                const select = $("#bookId");
                select.empty().append('<option value="">-- Chọn sách --</option>');
                
                if (response.data && response.data.length > 0) {
                    response.data.forEach(function(book) {
                        select.append(`<option value="${book.id}">${book.title}</option>`);
                    });
                }
            });
    }

    function loadBookInfo() {
        if (!selectedBookId) {
            selectedBookId = $("#bookId").val();
        }

        if (!selectedBookId) return;

        api.get(`/books/${selectedBookId}`)
            .done(function(response) {
                const book = response.data;
                $("#selectedBookTitle").text(book.title);
                $("#selectedBookAuthor").text(book.author || "Tác giả không xác định");
                $("#selectedBookPrice").text(formatCurrency(book.price || 0));
                $("#bookCover").attr("src", book.coverImage || "/assets/images/default-book.jpg");
                $("#selectedBookInfo").show();
            });
    }

    function changeBook() {
        selectedBookId = null;
        selectedRating = 0;
        $("#rating").val(0);
        updateStarDisplay(0);
        $("#bookSelectionGroup").show();
        $("#selectedBookInfo").hide();
        $("#changeBookBtn").hide();
        $("#reviewForm")[0].reset();
        $("#titleCount").text("0/100");
        $("#contentCount").text("0/1000");
        updateRatingText();
    }

    function updateStarDisplay(rating) {
        $("#starRating .star").each(function() {
            if ($(this).data("rating") <= rating) {
                $(this).addClass("active");
            } else {
                $(this).removeClass("active");
            }
        });
    }

    function updateRatingText() {
        const texts = {
            0: "Vui lòng chọn xếp hạng",
            1: "Rất tệ",
            2: "Không tốt",
            3: "Tạm được",
            4: "Rất tốt",
            5: "Tuyệt vời!"
        };
        $("#ratingText").text(texts[selectedRating]);
    }

    function validateForm() {
        if (!selectedBookId) {
            showAlert("danger", "Vui lòng chọn sách", "alertContainer");
            return false;
        }

        if (selectedRating === 0) {
            showAlert("danger", "Vui lòng chọn xếp hạng", "alertContainer");
            return false;
        }

        if (!$("#title").val().trim()) {
            showAlert("danger", "Vui lòng nhập tiêu đề đánh giá", "alertContainer");
            return false;
        }

        if (!$("#content").val().trim()) {
            showAlert("danger", "Vui lòng nhập nội dung đánh giá", "alertContainer");
            return false;
        }

        if ($("#content").val().trim().length < 20) {
            showAlert("danger", "Nội dung đánh giá cần có ít nhất 20 ký tự", "alertContainer");
            return false;
        }

        return true;
    }

    function showAlert(type, message, containerId) {
        const alertHTML = `
            <div class="alert alert-${type === 'danger' ? 'danger' : type} alert-dismissible fade show" role="alert" style="border-radius: 8px;">
                <i class="fas fa-${type === 'danger' ? 'exclamation-circle' : 'check-circle'} me-2"></i>${message}
                <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
            </div>
        `;
        $("#" + containerId).html(alertHTML);
    }

    function formatCurrency(value) {
        return new Intl.NumberFormat('vi-VN', {
            style: 'currency',
            currency: 'VND'
        }).format(value);
    }
</script>
